Interface Option<T>

An Option jonad.

Represents a value that may or may not be present. The left-side value is Some, while the right-side value is None (represented by the value null).

interface Option<T> {
    andThen: (<U>(mapper: ((value: T) => Option<U>)) => Option<U>);

    isNone: (() => boolean);
    isRight: (() => boolean);
    isSome: (() => boolean);
    leftOr: ((fallback: T | ((right: null) => T)) => T);
    leftOrAsync: ((fallback: T | ((right: null) => Promise<T>)) => Promise<T>);

    mapAsync: (<U>(mapper: ((value: T) => Promise<U>)) => Promise<Option<U>>);
    mapLeft: (<V>(mapper: ((value: T) => V)) => Either<V, null>);
    mapLeftAsync: (<V>(mapper: ((value: T) => Promise<V>)) => Promise<Either<V, null>>);

    matchAsync: (<V>(onLeft: ((value: T) => Promise<V>), onRight: ((value: null) => Promise<V>)) => Promise<V>);
