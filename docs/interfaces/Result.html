Interface Result<V, E>

A Result jonad, a subtype of Either.

Represents a value ("Ok") or an error ("Err"). Associates a set of utility functions to work with the jonad.

interface Result<V, E> {
    andThen: (<T>(mapper: ((value: V) => Result<T, E>)) => Result<T, E>);
    andThenAsync: (<T>(mapper: ((value: V) => Promise<Result<T, E>>)) => Promise<Result<T, E>>);
    getLeftOrThrow: (() => V);
    getRightOrThrow: (() => E);
    isErr: (() => boolean);
    isLeft: (() => boolean);
    isOk: (() => boolean);
    isRight: (() => boolean);
    leftOr: ((fallback: V | ((right: E) => V)) => V);
    leftOrAsync: ((fallback: V | ((right: E) => Promise<V>)) => Promise<V>);
    map: (<T>(mapper: ((value: V) => T)) => Result<T, E>);
    mapAsync: (<T>(mapper: ((value: V) => Promise<T>)) => Promise<Result<T, E>>);
    mapErr: (<T>(mapper: ((error: E) => T)) => Result<V, T>);
    mapErrAsync: (<T>(mapper: ((error: E) => Promise<T>)) => Promise<Result<V, T>>);
    mapLeft: (<V>(mapper: ((value: V) => V)) => Either<V, E>);
    mapLeftAsync: (<V>(mapper: ((value: V) => Promise<V>)) => Promise<Either<V, E>>);
    mapRight: (<V>(mapper: ((value: E) => V)) => Either<V, V>);
    mapRightAsync: (<V>(mapper: ((value: E) => Promise<V>)) => Promise<Either<V, V>>);
    match: (<V>(onLeft: ((value: V) => V), onRight: ((value: E) => V)) => V);
    matchAsync: (<V>(onLeft: ((value: V) => Promise<V>), onRight: ((value: E) => Promise<V>)) => Promise<V>);
    rightOr: ((fallback: E | ((left: V) => E)) => E);
    rightOrAsync: ((fallback: E | ((left: V) => Promise<E>)) => Promise<E>);
    tapLeft: ((callback: ((value: V) => void)) => Either<V, E>);
    tapLeftAsync: ((callback: ((value: V) => Promise<void>)) => Promise<Either<V, E>>);
    tapRight: ((callback: ((value: E) => void)) => Either<V, E>);
    tapRightAsync: ((callback: ((value: E) => Promise<void>)) => Promise<Either<V, E>>);
    valueOr: ((fallback: V | ((error: E) => V)) => V);
    valueOrAsync: ((fallback: V | ((error: E) => Promise<V>)) => Promise<V>);
}
